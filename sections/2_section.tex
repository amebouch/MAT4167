\section{Preliminary Notions}
\medskip 
{\color{gray}\hrule}
\medskip 
\subsection{Number fields and ring of integers}
\begin{flushleft}
    \quad\textbf{Definition:} A number field is a subfield of $\mathbb{C}$ that is finite-dimensional as a $\mathbb{Q}$-vector space.\\
    \quad\textbf{Notation:} Let $\alpha \in \mathbb{C}$, we define 
    \[
    \mathbb{Q}(\alpha) = \left\{ \frac{f(\alpha)}{g(\alpha)} \,\middle|\, f,g \in \mathbb{Q}[X],\ g(\alpha) \neq 0 \right\}.
    \]
    \quad\textbf{Definition:} Let K be a number field. The set of algebraic integers in K is called the ring of integers of K, denoted by $\mathcal{O}_K$.
\end{flushleft}

\subsection{Ideals and prime ideals}
\begin{flushleft}
    \quad\textbf{Definition:} Let $R$ be a ring. A subset $I$ of $R$ is called an \textit{ideal} of $R$ if it satisfies the following conditions:
    
    \begin{enumerate}
        \item $I$ is an additive subgroup of $(R, +)$;
        \item For every $r \in R$, $rI \subseteq I$ and $Ir \subseteq I$.
    \end{enumerate}
    
    Here, $rI := \{ ra : a \in I \}$ and $Ir := \{ ar : a \in I \}$.
\end{flushleft}
\begin{flushleft}
    \quad\textbf{Definition:} Let R be a commutative ring. An ideal $I$ of $R$ is called a \textit{prime ideal} if $I$ is proper and satisfies the following condition:
    \[
    \forall a, b \in R, ab \in I \Rightarrow a \in I or b \in I.
    \]
\end{flushleft}

\subsection{Reduction modulo p}
\begin{flushleft}
    \quad\textbf{Definition:} Let $K = \mathbb{Q}(\theta)$ be the number field, where $\theta$ has minimal polynomial $f(x) \in \mathbb{Z}[x]$ and let $p$ be a prime number. The reduction modulo (p) of f(x) is the polynomial obtained by reducing each coefficient of $f(x)$ modulo $p$. We obtain a polynomial
    \[
    \overline{f}(x) \in \mathbb{F}_p[x].
    \]
    Doing a reduction modulo $p$ allows one to study the factorization of $p$ in $\mathcal{O}_K$ via the factorization of $\overline{f}(x)$ in $\mathbb{F}_p[x]$. In particular, the way $\overline{f}(x)$ factors into irreducible polynomials determines how the ideal $p$ decomposes into prime ideals in $\mathcal{O}_K$.
\end{flushleft}

\subsection{Monogenic Field}
\begin{flushleft}
    \quad\textbf{Definition:} Let K be an algebraic number field. K is monogenic if there exists a $\theta$ 
    in $\mathcal{O}_K$ such that 
    \[
    \mathcal{O}_K = \mathbb{Z} + \mathbb{Z}\theta + \cdots + \mathbb{Z}\theta^{n-1}
    \]
    with $[K:\mathbb{Q}] = n$ (the degree of $K$ over $\mathbb{Q}$).
\end{flushleft}
\bigskip
We are interested in factorizing ideals of a monogenic number field into a product of 
prime ideals. A theorem proved by Dedekind in 1878 provides us with such a method, by considering congruences between the prime Factorization of 
an element of K and a product of irreducible polynomials in $\mathbb{Z}/p\mathbb{Z}$.

\subsection{Link between Reduction modulo p and Ideal Factorization}
The key idea behind Dedekind's theorem is that the arithmetic of the ring of integers $\mathcal{O}_K$ can be studied via polynomial arithmetic modulo a prime number $p$.

\medskip 

Let $K=\mathbb{Q}(\theta)$ be a monogenic number field with minimal polynomial $f(x) \in \mathbb{Z}[x]$.
Reducing $f(x)$ modulo $p$, we obtain a polynomial $\overline{f}(x) \in \mathbb{F}_p[x]$.

\medskip 

The factorization of $\overline{f}(x)$ into irreducible polynomials reflects how the ideal $<p> \subset \mathbb{O}_K$ decomposes into prime ideals.
More precisely, each irreducible factor of $\overline{f}(x)$ produce a prime ideal of $\mathbb{O}_K$. Moreover, the multiplicity of the factor corresponds to the exponent in the ideal factorization.

\medskip 

This correspondance can be found explicitly in Dedekind's theorem, which can be found in the next section.
