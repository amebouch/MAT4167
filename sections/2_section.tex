\section{Factoring Primes in a Monogenic Number Field}
\bigskip
\end{center}

\begin{flushleft}
    \quad\textbf{Theorem:} Let K = Q(theta) be an algebraic number field with degree n, such that 
    \[
    \mathcal{O}_K = \mathbb{Z} + \mathbb{Z}\theta + \cdots + \mathbb{Z}\theta^{n-1}
    \]
     (i.e. it's a monogenic number field). 
    Let p be a rational prime, and let f(x) be the minimal polynomial of theta in Z[x].
    \medskip
    
     Define a natural map 
     \[
     \phi : \mathbb{Z}[x] \to \mathbb{Z}_p[x] 
     \]
     be the reduction modulo $p$ map. 
     \medskip
     Then for $f(x) \in \mathbb{Z}[x]$, we can write
     \[
     \phi(f(x)) = g_1^{e_1} \cdots g_r^{e_r},
     \]
     where each $g_i \in \mathbb{Z}_p[x]$ is distinct, monic, and irreducible, and each $e_i \in \mathbb{Z}$.
    Any monic polynomial in Z[x] can be written as this product. 
    Define $f_i(x)$ to be a monic polynomial of $\mathbb{Z}[x]$ such that $\phi(f_i(x)) = g_i$.
    Set 
    \[
    Pi = <p, f_i(\theta)>, i = 1,2,...,r 
    \]
    to be the principal ideal generated by $p$ and $f_i(\theta)$. Each of these $P_i$'s are distinct
    prime ideals of $\mathcal{O}_K$ where the ideal generated by p is equal to a product of these ideals. 
    In other words, we can write each of these ideals as follows
    \[
    <p>=P_1^{e_1} \cdots P_r^{e_r}.
    \]
    Define the norm of $P_i$ to be $p$ raised to the power of the corresponding $f_i$. We denote it as follows
    \[
    N(P_i)=p^{deg f_i}, i = 1,2,...,r.
    \]
     
\end{flushleft}
